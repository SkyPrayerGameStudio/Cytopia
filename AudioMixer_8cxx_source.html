551  ALuint effect = 0;
552 
553  /* Create the effect object and check if we can do EAX reverb. */
554  alGenEffects(1, &effect);
555 
556  alEffecti(effect, AL_EFFECT_TYPE, AL_EFFECT_ECHO);
557 
558  alEffectf(effect, AL_ECHO_DELAY, echo_properties.flEchoDelay);
559  alEffectf(effect, AL_ECHO_LRDELAY, echo_properties.flEchoLRDelay);
560  alEffectf(effect, AL_ECHO_DAMPING, echo_properties.flEchoDamping);
561  alEffectf(effect, AL_ECHO_FEEDBACK, echo_properties.flEchoFeedback);
562  alEffectf(effect, AL_ECHO_SPREAD, echo_properties.flEchoSpread);
563 
564  /* Check if an error occured, and clean up if so. */
565  ALenum err = alGetError();
566  if (err != AL_NO_ERROR)
567  {
568  if (alIsEffect(effect))
569  alDeleteEffects(1, &effect);
570 
571  throw AudioError{TRACE_INFO "Unable to add reverb effect to track."};
572  fprintf(stderr, "OpenAL error: %s\n", alGetString(err));
573  }
574 
575  /* Create the effect slot object. This is what "plays" an effect on sources
576  * that connect to it. */
577  alGenAuxiliaryEffectSlots(1, &track->effect_slot);
578 
579  /* Tell the effect slot to use the loaded effect object. Note that the this
580  * effectively copies the effect properties. You can modify or delete the
581  * effect object afterward without affecting the effect slot.
582  */
583  alAuxiliaryEffectSloti(track->effect_slot, AL_EFFECTSLOT_EFFECT, (ALint)effect);
584  assert(alGetError() == AL_NO_ERROR && "Failed to set effect slot");
585 
586  alDeleteEffects(1, &effect);
587 
588  //apply effect to source
589  alSource3i(track->source, AL_AUXILIARY_SEND_FILTER, (ALint)(track->effect_slot), 0, AL_FILTER_NULL);
590  assert(alGetError() == AL_NO_ERROR && "Failed to setup reverb for sound source send 0.");
591 
592  //play sound
593 
594  alSourcePlay(track->source);
595 
596  m_Playing.push_front(&track);
597  track->isPlaying = true;
598 }
